create or replace function disp_player_skill(play_id in number) return varchar as skill_name varchar(50); begin begin select s.name into skill_name from skill s,player p where s.id=p.skill_id and p.id=play_id; case skill_name when 'Batsmen' then skill_name:='Player is a Batsmen'; when 'Bowler' then skill_name:='Player is a Bowler'; end case; exception when case_not_found then skill_name:='Player is an All Rounder'; when no_data_found then skill_name:='No Such player'; end; return(skill_name); end; /
